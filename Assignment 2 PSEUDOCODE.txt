DECLARE str1 as string
DECLARE str2 as string
DECLARE charRel as Character
DECLARE charRev as character
DECLARE count as integer
DECLARE char1 as Character

str1 = ""
str2 = ""
charRel = ''
charRev = ''
count = 0
char1 = ''

OUTPUT "Enter a string: "
INPUT str1

OUTPUT "Enter a character to remove: "
INPUT charRev

OUTPUT "Enter a character to replace: "

FOR count = 1 to LEN(str1)
	char1 = MID(str1,count,1)
	IF char1 = charRev THEN
	        str2 = str2 + charRel
	ELSE
		str2 = str2 + char1
	ENDIF
NEXT

OUTPUT "The new string is: " + str2
