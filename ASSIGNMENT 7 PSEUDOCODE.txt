DECLARE s as string
DECLARE string1 as string 
DECLARE string2 as string
DECLARE final as string

s = ""
string1 = ""
string2 = ""
final = ""

OUTPUT "Enter a word for manipulation: "
INPUT s

IF LEN(s) < 2 THEN
	OUTPUT "The final string is: " + s
ELSE
	string1 = LEFT(s,2)
	string2 = RIGHT(s,2)
	final = string1 + string2
	OUTPUT " The new string is: " + final
ENDIF
