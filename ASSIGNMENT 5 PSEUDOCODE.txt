DECLARE str1 as string
DECLARE vowel as string
DECLARE vowelC as integer
DECLARE count as integer
DECLARE char1 as character

str1 = ""
vowel = "aeiou"
vowelC = 0
count = 0
char1 = ''

OUTPUT "Enter a string to find total number of vowels"
INPUT str1

FOR count = 1 to LEN(str1)
	char1 = MID(str1,count,1)
        IF LOCATE(vowel,char1) > 0 THEN
	    vowelC = vowelC + 1
        ENDIF
NEXT
OUTPUT "The total number of vowel in the string are: " + vowel
